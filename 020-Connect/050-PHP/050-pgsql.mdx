```php
<?php

$dsn = sprintf("host=%s port=%d dbname=%s user=%s password=%s",
  "<REGION>.sql.xata.sh",
  5432,
  "<DATABASE_NAME>:<BRANCH_NAME>",
  "<WORKSPACE_ID>",
  "<API_KEY>"
);

// Run query
$conn = pg_connect($dsn) or die("failed to connect.");
$result = pg_query($conn, "SELECT 1");
var_dump(pg_fetch_all($result));

/*

  [0]=>
  array(1) {
    ["?column?"]=>
    string(1) "1"
  }
}
*/
```
